class Solution {
public:
    vector<int> getRow(int rowIndex) {
        if(rowIndex == 0) {
            return vector<int>(1,1);
        }
        if(rowIndex == 1) {
            return vector<int>(2,1);
        }
        vector<int> res(1,1), last = getRow(rowIndex-1);
        int n = last.size();
        for(int i = 0; i+1 < n; i++) {
            res.push_back(last[i]+last[i+1]);
        }
        res.push_back(1);
        return res;
    }
};
